快速开始

edge-function.ts

import { Hono } from 'hono';
import { cors } from 'hono/cors';

const app = new Hono();

app.use('/*', cors());

app.get('/api/status', (c) => {
  return c.json({
    "status": "healthy",
    "region": c.req.geo?.region,
    "timestamp": Date.now()
  });
});

export default app.fetch;

npm install @edge/runtime
